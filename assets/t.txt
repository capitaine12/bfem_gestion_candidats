from PyQt5.QtWidgets import QWidget, QVBoxLayout, QPushButton
from PyQt5.QtCore import Qt


class NavigationMenu(QWidget):
    """Classe repr√©sentant le menu de navigation √† gauche."""
    def __init__(self):
        super().__init__()

        self.setFixedWidth(180)  # Largeur de la barre de navigation
        self.setObjectName("navigationMenu")  # Ajout du nom d'objet
        nav_layout = QVBoxLayout()

        # Cr√©ation des boutons du menu
        self.btn_dashboard = QPushButton("üè† Dashboard")
        self.btn_candidats = QPushButton("üìã Liste des candidats")
        self.btn_deliberation = QPushButton("‚öñ D√©lib√©ration")
        self.btn_statistiques = QPushButton("üìä Statistiques")
        self.btn_quitter = QPushButton("‚ùå Quitter")

        # Appliquer le style aux boutons
        btn_style = """
            QPushButton {
                padding: 12px;
                font-size: 14px;
                background: transparent;
                color: black;
                text-align: left;
                border: none;
            }
            QPushButton:hover {
                background-color: rgba(255, 255, 255, 0.2);
            }
        """

        # Ajouter les boutons au layout de navigation
        for btn in [self.btn_dashboard, self.btn_candidats, self.btn_deliberation, self.btn_statistiques, self.btn_quitter]:
            btn.setStyleSheet(btn_style)
            btn.setCursor(Qt.PointingHandCursor)
            nav_layout.addWidget(btn)

        self.setLayout(nav_layout)
        self.setStyleSheet("""
            background-color: red;
            border-right: 2px solid #1a2e40;
        """)
